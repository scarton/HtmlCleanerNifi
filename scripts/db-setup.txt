create database epa; 
grant all on epa.* to 'epa'@'localhost' identified by 'epa123';
flush privileges;

use epa;
CREATE TABLE `cases` (
  `respondent` varchar(1024) DEFAULT NULL,
  `url` varchar(1024) DEFAULT NULL,
  `description` varchar(16384) DEFAULT NULL,
  `ordertype` varchar(64) DEFAULT NULL,
  `date` date DEFAULT NULL,
  `terms` text DEFAULT NULL
) ENGINE=InnoDB DEFAULT CHARSET=UTF8;

\C utf8

LOAD DATA LOCAL INFILE "c:/data/EPA/EPA Data2.csv" INTO TABLE epa.cases
CHARACTER SET utf8
FIELDS TERMINATED BY ','
ENCLOSED BY '"'
LINES TERMINATED BY '\n'
IGNORE 1 LINES
(respondent,url,description,ordertype,@date)
set date = STR_TO_DATE(@date,'%c/%d/%Y');
